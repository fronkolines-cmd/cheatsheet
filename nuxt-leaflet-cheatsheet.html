Nuxt Leaflet Cheatsheet

A comprehensive guide to using Leaflet maps in Nuxt 3 applications

Installation

Install Package

npm install @nuxtjs/leaflet

Configure nuxt.config.ts

export default defineNuxtConfig({
  modules: ['@nuxtjs/leaflet'],
  leaflet: {
    markerCluster: true
  }
})

Draggable Marker

<script setup>
const markerPosition = ref([51.505, -0.09])

function onMarkerDrag(event) {
  markerPosition.value = [event.target.getLatLng().lat, event.target.getLatLng().lng]
}
</script>

<template>
  <LMap style="height: 500px" :zoom="13" :center="[51.505, -0.09]">
    <LTileLayer url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png" />
    <LMarker 
      :lat-lng="markerPosition" 
      :draggable="true"
      @dragend="onMarkerDrag"
    />
  </LMap>
</template>

GeoJSON with Options

<script setup>
const geojson = { /* your geojson data */ }

const options = {
  onEachFeature: (feature, layer) => {
    if (feature.properties && feature.properties.name) {
      layer.bindPopup(feature.properties.name)
    }
  },
  style: (feature) => {
    return {
      color: feature.properties.color || '#3388ff',
      weight: 2,
      opacity: 0.6
    }
  }
}
</script>

<template>
  <LGeoJson :geojson="geojson" :options="options" />
</template>

Composables

useLeafletMap

<script setup>
const map = useLeafletMap()

function flyToLocation(lat, lng) {
  if (map.value) {
    map.value.flyTo([lat, lng], 15)
  }
}

function getMapBounds() {
  if (map.value) {
    return map.value.getBounds()
  }
}
</script>

<template>
  <LMap ref="map">
    <!-- map content -->
  </LMap>
  <button @click="flyToLocation(51.505, -0.09)">Fly to Location</button>
</template>

useLeafletDisplay Options

<script setup>
const { mapRef } = useLeafletDisplay()

onMounted(() => {
  if (mapRef.value) {
    mapRef.value.setZoom(10)
    mapRef.value.panTo([51.505, -0.09])
  }
})
</script>

Map Instance Methods

<script setup>
const mapRef = ref(null)

function fitBounds() {
  const bounds = [
    [51.49, -0.08],
    [51.5, -0.06]
  ]
  mapRef.value?.leafletObject?.fitBounds(bounds)
}

function setView(lat, lng, zoom) {
  mapRef.value?.leafletObject?.setView([lat, lng], zoom)
}

function invalidateSize() {
  mapRef.value?.leafletObject?.invalidateSize()
}
</script>

<template>
  <LMap ref="mapRef">
    <!-- map content -->
  </LMap>
</template>
